# --description--

구골($10^{100}$)은 엄청난 숫자입니다: 1 뒤에 100개의 0이 붙은 수이고, $100^{100}$은 거의 상상할 수 없을 만큼 큽니다: 1 뒤에 200개의 0이 붙은 수입니다. 이 숫자들의 크기에도 불구하고, 각 숫자의 자릿수 합은 단지 1입니다.

`a`, `b` < `n`인 형태의 자연수 $a^b$를 고려할 때, 최대 자릿수 합은 얼마일까요?

function powerfulDigitSum(n) {
  function sumDigitsOfPower(numA, numB) {
    let digitsSum = 0;
    let number = power(numA, numB);
    while (number > 0n) {
      const digit = number % 10n;
      digitsSum += parseInt(digit, 10);
      number = number / 10n;
    }
    return digitsSum;
  }

  function power(numA, numB) {
    let sum = 1n;
    for (let b = 0; b < numB; b++) {
      sum = sum * BigInt(numA);
    }
    return sum;
  }

  const limit = n - 1;
  let maxDigitsSum = 0;
  for (let a = limit; a > 0; a--) {
    for (let b = limit; b > 0; b--) {
      const curDigitSum = sumDigitsOfPower(a, b);
      if (curDigitSum > maxDigitsSum) {
        maxDigitsSum = curDigitSum;
      }
    }
  }
  return maxDigitsSum;
}
