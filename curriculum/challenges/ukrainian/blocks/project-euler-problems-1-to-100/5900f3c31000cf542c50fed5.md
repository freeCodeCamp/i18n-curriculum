# --description--

Павук S сидить у одному куті прямокутного паралелепіпеда розміром 6 на 5 на 3, а муха F — у протилежному куті. Переміщаючись по поверхнях кімнати, найкоротша «пряма» відстань від S до F дорівнює 10, і шлях показано на діаграмі.

<img alt="діаграма шляху павука і мухи від одного кута прямокутного паралелепіпеда до протилежного" src="https://cdn-media-1.freecodecamp.org/project-euler/cuboid-route.png" style="background-color: white; padding: 10px; display: block; margin-right: auto; margin-left: auto; margin-bottom: 1.2rem;">

Однак для будь-якого прямокутного паралелепіпеда існує до трьох кандидатів на «найкоротший» шлях, і найкоротший маршрут не завжди має цілу довжину.

Можна показати, що існує рівно `2060` різних прямокутних паралелепіпедів, ігноруючи обертання, з цілими розмірами до максимальної величини M на M на M, для яких найкоротший маршрут має цілу довжину, коли M = 100. Це найменше значення M, при якому кількість розв’язків вперше перевищує дві тисячі; кількість розв’язків при M = 99 дорівнює `1975`.

Знайдіть найменше значення M, при якому кількість розв’язків вперше перевищує `n`.

```js
function cuboidRoute(n) {
  // Based on https://www.mathblog.dk/project-euler-86-shortest-path-cuboid/
  function getLength(a, b) {
    return Math.sqrt(a ** 2 + b ** 2);
  }

  let M = 2;
  let counter = 0;

  while (counter < n) {
    M++;
    for (let baseHeightWidth = 3; baseHeightWidth <= 2 * M; baseHeightWidth++) {
      const pathLength = getLength(M, baseHeightWidth);
      if (Number.isInteger(pathLength)) {
        if (baseHeightWidth <= M) {
          counter += Math.floor(baseHeightWidth / 2);
        } else {
          counter += 1 + M - Math.floor((baseHeightWidth + 1) / 2);
        }
      }
    }
  }

  return M;
}
```
