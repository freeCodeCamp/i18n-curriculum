```js
function factorial(n) {
```

```js
function factorial(n) {
  const arr = Array(n).fill().map((_, i) => i + 1);
  return arr.reduce((acc, curr) => acc * curr, 1);
}
```

```js
function factorial(n) {
  let result = 1;

  for (let i = n; i > 0; i--) {
    result *= i;
```

#### --answer--

```js
function factorial(n) {
  if (n === 0 || n === 1) {
    return 1;
  }
  return n * factorial(n - 1);
}
```
