# --description--

العدد الطبيعي، `N`، الذي يمكن كتابته كمجموع وحاصل ضرب لمجموعة معينة من رقمين طبيعيين على الأقل، $\\{a_1, a_2, \ldots , a_k\\}$ يُسمى عدد حاصل الجمع والضرب: $N = a_1 + a_2 + \cdots + a_k = a_1 × a_2 × \cdots × a_k$.

على سبيل المثال، 6 = 1 + 2 + 3 = 1 × 2 × 3.

لمجموعة بحجم `k`، نُسمّي أصغر N تحقق هذه الخاصية عدد حاصل الجمع والضرب الأدنى. أعداد حاصل الجمع والضرب الأدنى للمجموعات بحجم `k` = 2، 3، 4، 5، و6 هي كما يلي.

<div style='margin-left: 4em;'>
  <var>k</var>=2: 4 = 2 × 2 = 2 + 2<br>
  <var>k</var>=3: 6 = 1 × 2 × 3 = 1 + 2 + 3<br>
  <var>k</var>=4: 8 = 1 × 1 × 2 × 4 = 1 + 1 + 2 + 4<br>
  <var>k</var>=5: 8 = 1 × 1 × 2 × 2 × 2  = 1 + 1 + 2 + 2 + 2<br>
  <var>k</var>=6: 12 = 1 × 1 × 1 × 1 × 2 × 6 = 1 + 1 + 1 + 1 + 2 + 6
</div><br>

لذا، عندما 2 ≤ `k` ≤ 6، مجموع كل أعداد حاصل الجمع والضرب الأدنى هو 4 + 6 + 8 + 12 = 30؛ مع ملاحظة أن `8` تُحسب مرة واحدة فقط في المجموع.

في الواقع، بما أن المجموعة الكاملة لأعداد حاصل الجمع والضرب الأدنى لـ 2 ≤ `k` ≤ 12 هي $\\{4, 6, 8, 12, 15, 16\\}$، فإن المجموع هو `61`.

ما هو مجموع كل أعداد حاصل الجمع والضرب الأدنى لـ 2 ≤ `k` ≤ `limit`؟

function productSumNumbers(limit) {
  function getProductSums(curProduct, curSum, factorsCount, start) {
    const k = curProduct - curSum + factorsCount;
    if (k <= limit) {
      if (curProduct < minimalProductSums[k]) {
        minimalProductSums[k] = curProduct;
      }
      for (let i = start; i < Math.floor((limit / curProduct) * 2) + 1; i++) {
        getProductSums(curProduct * i, curSum + i, factorsCount + 1, i);
      }
    }
  }

  const minimalProductSums = new Array(limit + 1).fill(2 * limit);
  getProductSums(1, 1, 1, 2);

  const uniqueProductSums = [...new Set(minimalProductSums.slice(2))];

  let sum = 0;
  for (let i = 0; i < uniqueProductSums.length; i++) {
    sum += uniqueProductSums[i];
  }

  return sum;
}
