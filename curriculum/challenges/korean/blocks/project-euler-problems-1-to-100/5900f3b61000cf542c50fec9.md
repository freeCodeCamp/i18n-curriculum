# --description--

숫자 145는 각 자리 숫자의 팩토리얼 합이 145와 같다는 속성으로 잘 알려져 있습니다:

$$1! + 4! + 5! = 1 + 24 + 120 = 145$$

덜 알려졌지만 169는 169로 다시 연결되는 가장 긴 숫자 연쇄를 만들어냅니다; 실제로 존재하는 그러한 고리는 세 개뿐입니다:

$$\begin{align}
&169 → 363601 → 1454 → 169\\\\
&871 → 45361 → 871\\\\
&872 → 45362 → 872\\\\
\end{align}$$

모든 시작 숫자는 결국 고리에 갇히게 된다는 것을 증명하는 것은 어렵지 않습니다. 예를 들어,

$$\begin{align}
&69 → 363600 → 1454 → 169 → 363601\\ (→ 1454)\\\\
&78 → 45360 → 871 → 45361\\ (→ 871)\\\\
&540 → 145\\ (→ 145)\\\\
\end{align}$$

69로 시작하면 다섯 개의 중복되지 않는 항으로 이루어진 연쇄가 생성되지만, 백만 미만의 시작 숫자 중 가장 긴 중복되지 않는 연쇄는 60개 항입니다.

`n` 미만의 시작 숫자를 가진 연쇄 중 정확히 60개의 중복되지 않는 항을 포함하는 연쇄는 몇 개일까요?

function digitFactorialChains(n) {
  function sumDigitsFactorials(number) {
    let sum = 0;
    while (number > 0) {
      sum += factorials[number % 10];
      number = Math.floor(number / 10);
    }
    return sum;
  }

  const factorials = [1];
  for (let i = 1; i < 10; i++) {
    factorials.push(factorials[factorials.length - 1] * i);
  }

  const sequences = {
    169: 3,
    871: 2,
    872: 2,
    1454: 3,
    45362: 2,
    45461: 2,
    3693601: 3
  };
  let result = 0;

  for (let i = 2; i < n; i++) {
    let curNum = i;
    let chainLength = 0;
    const curSequence = [];
    while (curSequence.indexOf(curNum) === -1) {
      curSequence.push(curNum);
      curNum = sumDigitsFactorials(curNum);
      chainLength++;
      if (sequences.hasOwnProperty(curNum) > 0) {
        chainLength += sequences[curNum];
        break;
      }
    }
    if (chainLength === 60) {
      result++;
    }
    for (let j = 1; j < curSequence.length; j++) {
      sequences[curSequence[j]] = chainLength - j;
    }
  }
  return result;
}
