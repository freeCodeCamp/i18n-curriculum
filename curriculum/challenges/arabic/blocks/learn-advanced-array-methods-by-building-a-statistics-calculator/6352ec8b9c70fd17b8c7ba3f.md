# --description--

الآن عليك فرز القيم بشكل صحيح. اربط طريقة `.sort()` باستدعاء `Object.keys()` الخاص بك.

بالنسبة لاستدعاء الراجع، ستحتاج إلى استخدام كائن `counts` لمقارنة قيم كل مفتاح. يمكنك استخدام المعلمتين `a` و `b` للوصول إلى المفاتيح. ثم، أعد قيمة `counts[b]` ناقص قيمة `counts[a]`.

أخيرًا، استخدم ترميز الأقواس للوصول إلى العنصر الأول في المصفوفة لإكمال متغير `highest` الخاص بك.

const getMean = (array) => array.reduce((acc, el) => acc + el, 0) / array.length;

const getMedian = (array) => {
  const sorted = array.toSorted((a, b) => a - b);
  const median =
    sorted.length % 2 === 0
      ? getMean([sorted[sorted.length / 2], sorted[sorted.length / 2 - 1]])
      : sorted[Math.floor(sorted.length / 2)];
  return median;
}
