# --description--

Можна побачити, що число 125874 і його подвійне 251748 містять точно ті ж цифри, але в іншому порядку.

Знайдіть найменше додатне ціле число, яке при множенні на цілі числа $\\{2, 3, \ldots, n\\}$ містить ті ж цифри.

```js
function permutedMultiples(n) {
    const isPermutation = (a, b) =>
        a.length !== b.length
            ? false
            : a.split('').sort().join() === b.split('').sort().join();


    let start = 1;
    let found = false;
    let result = 0;

    while (!found) {
        start *= 10;
        for (let i = start; i < start * 10 / n; i++) {
            found = true;
            for (let j = 2; j <= n; j++) {
                if (!isPermutation(i + '', j * i + '')) {
                    found = false;
                    break;
                }
            }
            if (found) {
                result = i;
                break;
            }
        }
    }

    return result;
}
```
