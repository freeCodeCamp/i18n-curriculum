# --description--

Число 1406357289 є пандигітальним числом від 0 до 9, оскільки воно складається з кожної цифри від 0 до 9 у певному порядку, але також має досить цікаву властивість ділення підрядків.

Нехай $d_1$ — це $1^{ша}$ цифра, $d_2$ — $2^{га}$ цифра і так далі. Таким чином, ми маємо:

- ${d_2}{d_3}{d_4} = 406$ ділиться на 2
- ${d_3}{d_4}{d_5} = 063$ ділиться на 3
- ${d_4}{d_5}{d_6} = 635$ ділиться на 5
- ${d_5}{d_6}{d_7} = 357$ ділиться на 7
- ${d_6}{d_7}{d_8} = 572$ ділиться на 11
- ${d_7}{d_8}{d_9} = 728$ ділиться на 13
- ${d_8}{d_9}{d_{10}} = 289$ ділиться на 17

Знайдіть суму всіх пандигітальних чисел від 0 до `n`, у яких підрядки задовольняють `n - 2` цих властивостей ділення.

**Примітка:** У результаті слід враховувати пандигітальні числа, що починаються з `0`.

function substringDivisibility(n) {
  function isSubDivisible(digits) {
    const factors = [2, 3, 5, 7, 11, 13, 17];

    for (let i = 1; i < digits.length - 2; i++) {
      const subNumber = digits[i] * 100 + digits[i + 1] * 10 + digits[i + 2];
      if (subNumber % factors[i - 1] !== 0) {
        return false;
      }
    }
    return true;
  }

  function heapsPermutations(k, digits, conditionCheck, results) {
    if (k === 1) {
      if (conditionCheck(digits)) {
        const number = parseInt(digits.join(''), 10);
        results.push(number);
      }
      return;
    }

    heapsPermutations(k - 1, digits, conditionCheck, results);

    for (let i = 0; i < k - 1; i++) {
      if (k % 2 === 0) {
        [digits[i], digits[k - 1]] = [digits[k - 1], digits[i]];
      } else {
        [digits[0], digits[k - 1]] = [digits[k - 1], digits[0]];
      }
      heapsPermutations(k - 1, digits, conditionCheck, results);
    }
    return;
  }

  const allowedDigits = [...new Array(n + 1).keys()];
  const divisiblePandigitals = [];
  heapsPermutations(
    allowedDigits.length,
    allowedDigits,
    isSubDivisible,
    divisiblePandigitals
  );

  let sum = 0;
  for (let i = 0; i < divisiblePandigitals.length; i++) {
    sum += divisiblePandigitals[i];
  }

  return sum;
}
