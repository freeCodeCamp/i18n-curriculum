# --description--

5人の船乗りが島で難破し、昼間に大量のココナッツを集めました。

その夜、最初の船乗りが目を覚まし、早めに自分の分け前を取ろうと考え、ココナッツの山を5つに均等に分けようとしましたが、1つ余ってしまいました。そこでその1つをサルに投げ与え、5つに均等に分けられたうちの「自分の」1つの山を隠し、残りの4つの山をまとめて再び1つの見えるココナッツの山にしてから寝ました。

話を簡単にすると、夜の間に5人の船乗りが順番に起きて同じ行動を繰り返します。ココナッツの山を5つに分け、1つ余るココナッツをサルに渡し、残りの4つの山をまとめて1つの山にします。

朝になると（夜の間に5人の船乗りがそれぞれこっそり行った行動の後）、残ったココナッツは5つに均等に分けられ、余りなく船乗りたちに分配されます。（朝はサルに渡すココナッツはありません。）

# --instructions--

`N`人の船乗りのために、昼間に集められた最初のココナッツの山の最小の可能な大きさを返す関数を作成してください。**注意：** この話は、昼間に任意の数のココナッツを集めたり、山を何度も分割したりすることが物語の時間内で可能である世界で語られているため、数学的な影響はありません。

function splitCoconuts(intSailors) {
  let intNuts = intSailors;
  let result = splitCoconutsHelper(intNuts, intSailors);
  while (!result) {
    intNuts += 1;
    result = splitCoconutsHelper(intNuts, intSailors);
  }

  return intNuts;
}

function splitCoconutsHelper(intNuts, intSailors, intDepth) {
  const nDepth = intDepth !== undefined ? intDepth : intSailors;
  const portion = Math.floor(intNuts / intSailors);
  const remain = intNuts % intSailors;

  if (portion <= 0 || remain !== (nDepth ? 1 : 0)) {
    return null;
  }

  if (nDepth) {
    return splitCoconutsHelper(
      intNuts - portion - remain, intSailors, nDepth - 1
    );
  }

  return intNuts;
}
