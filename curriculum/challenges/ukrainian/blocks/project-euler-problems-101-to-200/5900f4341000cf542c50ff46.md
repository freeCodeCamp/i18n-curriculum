# --description--

Три кола однакового радіуса розміщені всередині більшого кола так, що кожна пара кіл дотикається одне до одного, і внутрішні кола не перекриваються. Існує чотири незаповнені "горизонтальні розриви", які потрібно ітеративно заповнювати більшою кількістю дотичних кіл.

<img alt="a diagram of non-overlapping circles" src="https://cdn-media-1.freecodecamp.org/project-euler/199-circles-in-circles.gif" style="background-color: white; padding: 10px; display: block; margin-right: auto; margin-left: auto; margin-bottom: 1.2rem;">

На кожній ітерації в кожен розрив поміщається максимально можливе коло, що створює більше розривів для наступної ітерації. Після 3 ітерацій (на зображенні) існує 108 розривів, а дріб області, яка не покрита колами, становить 0.06790342, округлено до восьми десяткових знаків.

Який дріб області не покритий колами після `n` ітерацій? Вкажіть відповідь, округлену до восьми десяткових знаків у форматі x.xxxxxxxx .

function iterativeCirclePacking(n) {
  let k1 = 1;
  let k0 = k1 * (3 - 2 * Math.sqrt(3));
  let a0 = 1 / (k0 * k0);
  let a1 = 3 / (k1 * k1);
  a1 += 3 * getArea(k0, k1, k1, n);
  a1 += getArea(k1, k1, k1, n);
  let final = ((a0 - a1) / a0).toFixed(8);

  return parseFloat(final);
  function getArea(k1, k2, k3, depth) {
      if (depth == 0) return 0.0;
      let k4 = k1 + k2 + k3 + 2 * Math.sqrt(k1 * k2 + k2 * k3 + k3 * k1);
      let a = 1 / (k4 * k4);
      a += getArea(k1, k2, k4, depth - 1);
      a += getArea(k2, k3, k4, depth - 1);
      a += getArea(k3, k1, k4, depth - 1);
      return a;
  }
}
