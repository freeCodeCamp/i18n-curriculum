# --description--

Inajulikana vizuri kwamba ikiwa mzizi wa mraba wa nambari asilia si nambari kamili, basi ni nambari isiyo na mantiki. Upanuzi wa tarakimu ya desimali wa mizizi kama hiyo ya mraba ni usio na mwisho bila muundo wowote unaojirudia kabisa.

Mzizi wa mraba wa mbili ni `1.41421356237309504880...`, na jumla ya tarakimu za desimali mia moja za kwanza ni `475`.

Kwa nambari asilia `n` za kwanza, pata jumla ya jumla za tarakimu za desimali mia moja za kwanza kwa mizizi yote isiyo na mantiki ya mraba.

function sqrtDigitalExpansion(n) {
  function sumDigits(number) {
    let sum = 0;
    while (number > 0n) {
      let digit = number % 10n;
      sum += parseInt(digit, 10);
      number = number / 10n;
    }
    return sum;
  }

  function power(numberA, numberB) {
    let result = 1n;
    for (let b = 0; b < numberB; b++) {
      result = result * BigInt(numberA);
    }
    return result;
  }

  // Based on http://www.afjarvis.staff.shef.ac.uk/maths/jarvisspec02.pdf
  function expandSquareRoot(number, numDigits) {
    let a = 5n * BigInt(number);
    let b = 5n;
    const boundaryWithNeededDigits = power(10, numDigits + 1);

    while (b < boundaryWithNeededDigits) {
      if (a >= b) {
        a = a - b;
        b = b + 10n;
      } else {
        a = a * 100n;
        b = (b / 10n) * 100n + 5n;
      }
    }
    return b / 100n;
  }

  let result = 0;
  let nextPerfectRoot = 1;
  const requiredDigits = 100;
  for (let i = 1; i <= n; i++) {
    if (nextPerfectRoot ** 2 === i) {
      nextPerfectRoot++;
      continue;
    }
    result += sumDigits(expandSquareRoot(i, requiredDigits));
  }

  return result;
}
