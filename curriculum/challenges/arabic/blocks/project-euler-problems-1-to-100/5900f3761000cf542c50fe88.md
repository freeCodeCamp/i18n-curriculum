# --description--

ثلاثي فيثاغورس هو مجموعة من ثلاثة أعداد طبيعية، حيث `a` &lt; `b` &lt; `c`، وتحقق العلاقة التالية،

<div style='text-align: center;'><var>a</var><sup>2</sup> + <var>b</var><sup>2</sup> = <var>c</var><sup>2</sup></div>

على سبيل المثال، 3<sup>2</sup> + 4<sup>2</sup> = 9 + 16 = 25 = 5<sup>2</sup>.

يوجد ثلاثي فيثاغورس واحد فقط بحيث يكون `a` + `b` + `c` = 1000. جد حاصل ضرب `abc` بحيث `a` + `b` + `c` = `n`.

function specialPythagoreanTriplet(n) {
  for (let a = 1; a <= n / 2; a++) {
    for (let b = a + 1; b <= n / 2; b++) {
      const c = Math.sqrt(a * a + b * b);

      if ((a + b + c) % n == 0) {
        const factor = n / (a + b + c);

        return (a * b * c) * (factor ** 3);
      }
    }
  }
}
