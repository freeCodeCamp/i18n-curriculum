# --description--

العدد googol ($10^{100}$) هو عدد ضخم: واحد متبوع بمئة صفر؛ والعدد $100^{100}$ أكبر بشكل لا يُتصور تقريبًا: واحد متبوع بمئتي صفر. رغم حجمهما، مجموع الأرقام في كل عدد هو فقط 1.

بالنظر إلى الأعداد الطبيعية على الشكل $a^b$، حيث `a` و `b` أصغر من `n`، ما هو أكبر مجموع للأرقام؟

function powerfulDigitSum(n) {
  function sumDigitsOfPower(numA, numB) {
    let digitsSum = 0;
    let number = power(numA, numB);
    while (number > 0n) {
      const digit = number % 10n;
      digitsSum += parseInt(digit, 10);
      number = number / 10n;
    }
    return digitsSum;
  }

  function power(numA, numB) {
    let sum = 1n;
    for (let b = 0; b < numB; b++) {
      sum = sum * BigInt(numA);
    }
    return sum;
  }

  const limit = n - 1;
  let maxDigitsSum = 0;
  for (let a = limit; a > 0; a--) {
    for (let b = limit; b > 0; b--) {
      const curDigitSum = sumDigitsOfPower(a, b);
      if (curDigitSum > maxDigitsSum) {
        maxDigitsSum = curDigitSum;
      }
    }
  }
  return maxDigitsSum;
}
