# --description--

Implementieren Sie explizit die schriftliche Multiplikation.

Dies ist ein möglicher Ansatz für eine Ganzzahlarithmetik mit beliebiger Genauigkeit.

# --instructions--

Schreiben Sie eine Funktion, die zwei Strings großer Zahlen als Parameter entgegennimmt. Ihre Funktion soll das Produkt dieser beiden großen Zahlen als String zurückgeben.

**Hinweis:** In JavaScript sind arithmetische Operationen bei großen Zahlen ungenau, daher müssen Sie die präzise Multiplikation selbst implementieren.

function mult(strNum1, strNum2) {
    var a1 = strNum1.split("").reverse();
    var a2 = strNum2.toString().split("").reverse();
    var aResult = new Array;

    for ( var iterNum1 = 0; iterNum1 < a1.length; iterNum1++ ) {
        for ( var iterNum2 = 0; iterNum2 < a2.length; iterNum2++ ) {
            var idxIter = iterNum1 + iterNum2;    // Get the current array position.
            aResult[idxIter] = a1[iterNum1] * a2[iterNum2] + ( idxIter >= aResult.length ? 0 : aResult[idxIter] );

            if ( aResult[idxIter] > 9 ) {    // Carrying
                aResult[idxIter + 1] = Math.floor( aResult[idxIter] / 10 ) + ( idxIter + 1 >= aResult.length ? 0 : aResult[idxIter + 1] );
                aResult[idxIter] %= 10;
            }
        }
    }
    return aResult.reverse().join("");
}
