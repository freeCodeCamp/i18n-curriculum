# --description--

Eine vollkommene Zahl ist eine Zahl, bei der die Summe ihrer echten Teiler genau der Zahl selbst entspricht. Zum Beispiel ist die Summe der echten Teiler von 28 gleich 1 + 2 + 4 + 7 + 14 = 28, was bedeutet, dass 28 eine vollkommene Zahl ist.

Eine Zahl `n` wird als defizient bezeichnet, wenn die Summe ihrer echten Teiler kleiner als `n` ist, und als übermäßig, wenn diese Summe `n` übersteigt.

Da 12 die kleinste übermäßige Zahl ist, nämlich 1 + 2 + 3 + 4 + 6 = 16, ist die kleinste Zahl, die als Summe zweier übermäßiger Zahlen geschrieben werden kann, 24. Durch mathematische Analyse lässt sich zeigen, dass alle ganzen Zahlen größer als 28123 als Summe zweier übermäßiger Zahlen dargestellt werden können. Diese obere Grenze kann jedoch durch Analyse nicht weiter reduziert werden, obwohl bekannt ist, dass die größte Zahl, die nicht als Summe zweier übermäßiger Zahlen ausgedrückt werden kann, kleiner als diese Grenze ist.

Finden Sie die Summe aller positiven ganzen Zahlen &lt;= `n`, die nicht als Summe zweier übermäßiger Zahlen geschrieben werden können.

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
