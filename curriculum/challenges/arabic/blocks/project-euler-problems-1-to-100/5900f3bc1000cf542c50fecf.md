# --description--

من المعروف أنه إذا لم يكن الجذر التربيعي لعدد طبيعي عددًا صحيحًا، فإنه يكون عددًا غير نسبي. التوسع العشري لمثل هذه الجذور التربيعية لا نهائي ولا يحتوي على أي نمط متكرر على الإطلاق.

الجذر التربيعي للعدد اثنين هو `1.41421356237309504880...`، ومجموع الأرقام الرقمية لأول مئة رقم عشري هو `475`.

لأول `n` أعداد طبيعية، احسب مجموع المجاميع الرقمية لأول مئة رقم عشري لجميع الجذور التربيعية غير النسبية.

function sqrtDigitalExpansion(n) {
  function sumDigits(number) {
    let sum = 0;
    while (number > 0n) {
      let digit = number % 10n;
      sum += parseInt(digit, 10);
      number = number / 10n;
    }
    return sum;
  }

  function power(numberA, numberB) {
    let result = 1n;
    for (let b = 0; b < numberB; b++) {
      result = result * BigInt(numberA);
    }
    return result;
  }

  // Based on http://www.afjarvis.staff.shef.ac.uk/maths/jarvisspec02.pdf
  function expandSquareRoot(number, numDigits) {
    let a = 5n * BigInt(number);
    let b = 5n;
    const boundaryWithNeededDigits = power(10, numDigits + 1);

    while (b < boundaryWithNeededDigits) {
      if (a >= b) {
        a = a - b;
        b = b + 10n;
      } else {
        a = a * 100n;
        b = (b / 10n) * 100n + 5n;
      }
    }
    return b / 100n;
  }

  let result = 0;
  let nextPerfectRoot = 1;
  const requiredDigits = 100;
  for (let i = 1; i <= n; i++) {
    if (nextPerfectRoot ** 2 === i) {
      nextPerfectRoot++;
      continue;
    }
    result += sumDigits(expandSquareRoot(i, requiredDigits));
  }

  return result;
}
