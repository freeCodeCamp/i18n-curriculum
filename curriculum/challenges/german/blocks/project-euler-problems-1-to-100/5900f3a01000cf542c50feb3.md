# --description--

Man sieht, dass die Zahl 125874 und ihr Doppeltes, 251748, genau dieselben Ziffern enthalten, aber in unterschiedlicher Reihenfolge.

Finden Sie die kleinste positive ganze Zahl, die, multipliziert mit den ganzen Zahlen $\\{2, 3, \ldots, n\\}$, dieselben Ziffern enthält.

```js
function permutedMultiples(n) {
    const isPermutation = (a, b) =>
        a.length !== b.length
            ? false
            : a.split('').sort().join() === b.split('').sort().join();


    let start = 1;
    let found = false;
    let result = 0;

    while (!found) {
        start *= 10;
        for (let i = start; i < start * 10 / n; i++) {
            found = true;
            for (let j = 2; j <= n; j++) {
                if (!isPermutation(i + '', j * i + '')) {
                    found = false;
                    break;
                }
            }
            if (found) {
                result = i;
                break;
            }
        }
    }

    return result;
}
```
