# --description--

يُنشأ كسر عشري غير نسبي بدمج الأعداد الصحيحة الموجبة:

0.12345678910**1**112131415161718192021...

يمكن ملاحظة أن الرقم الثاني عشر من الجزء الكسري هو 1.

d<sub>1</sub> × d<sub>10</sub> × d<sub>100</sub> × d<sub>1000</sub> × d<sub>10000</sub> × d<sub>100000</sub> × d<sub>1000000</sub>

function champernownesConstant(n) {
  let fractionalPart = '';
  for (let i = 0; fractionalPart.length <= n; i++) {
    fractionalPart += i.toString();
  }

  let product = 1;
  for (let i = 0; i < n.toString().length; i++) {
    const index = 10 ** i;
    product *= parseInt(fractionalPart[index], 10);
  }

  return product;
}
