# --description--

Сума квадратів перших десяти натуральних чисел дорівнює,

<div style='text-align: center;'>1<sup>2</sup> + 2<sup>2</sup> + ... + 10<sup>2</sup> = 385</div>

Квадрат суми перших десяти натуральних чисел дорівнює,

<div style='text-align: center;'>(1 + 2 + ... + 10)<sup>2</sup> = 55<sup>2</sup> = 3025</div>

Отже, різниця між сумою квадратів перших десяти натуральних чисел і квадратом суми дорівнює 3025 − 385 = 2640.

Знайдіть різницю між сумою квадратів перших `n` натуральних чисел і квадратом суми.

const sumSquareDifference = (number)=>{
  let squareOfSum = Math.pow(sumOfArithmeticSeries(1,1,number),2);
  let sumOfSquare = sumOfSquareOfNumbers(number);
 return squareOfSum - sumOfSquare;
}

function sumOfArithmeticSeries(a,d,n){
  return (n/2)*(2*a+(n-1)*d);
}

function sumOfSquareOfNumbers(n){
 return (n*(n+1)*(2*n+1))/6;
}
