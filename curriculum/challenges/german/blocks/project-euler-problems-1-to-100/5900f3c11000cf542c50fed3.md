Ein Spieler startet auf dem Feld GO und addiert die Augenzahlen von zwei sechsseitigen Würfeln, um die Anzahl der Felder zu bestimmen, die er im Uhrzeigersinn vorrückt. Ohne weitere Regeln würden wir erwarten, dass jedes Feld mit gleicher Wahrscheinlichkeit besucht wird: 2,5 %. Allerdings verändern das Landen auf G2J (Gehe ins Gefängnis), CC (Community Chest) und CH (Chance) diese Verteilung.

Zusätzlich zu G2J und je einer Karte von CC und CH, die den Spieler direkt ins Gefängnis schicken, gilt: Wenn ein Spieler dreimal hintereinander einen Pasch würfelt, rückt er nicht das Ergebnis des dritten Wurfs vor, sondern geht direkt ins Gefängnis.

Zu Beginn des Spiels werden die CC- und CH-Karten gemischt. Wenn ein Spieler auf CC oder CH landet, zieht er die oberste Karte vom jeweiligen Stapel und folgt den Anweisungen. Danach wird die Karte unter den Stapel zurückgelegt. Es gibt sechzehn Karten pro Stapel, aber für diese Aufgabe interessieren uns nur die Karten, die eine Bewegung anordnen; alle anderen Anweisungen werden ignoriert, und der Spieler bleibt auf dem CC-/CH-Feld.

    <li>Vorrücken zum nächsten R (Eisenbahngesellschaft)</li>
    <li>Vorrücken zum nächsten R</li>
    <li>Vorrücken zum nächsten U (Versorgungsunternehmen)</li>
    <li>Gehe 3 Felder zurück.</li>
  </ol>
</ul>

Der Kern dieser Aufgabe betrifft die Wahrscheinlichkeit, ein bestimmtes Feld zu besuchen. Das heißt, die Wahrscheinlichkeit, nach einem Wurf auf diesem Feld zu landen. Deshalb sollte klar sein, dass mit Ausnahme von G2J, für das die Wahrscheinlichkeit, darauf zu landen, null ist, die CH-Felder die niedrigsten Wahrscheinlichkeiten haben, da 5 von 8 Karten eine Bewegung zu einem anderen Feld anordnen. Uns interessiert das Feld, auf dem der Spieler nach jedem Wurf tatsächlich landet. Wir unterscheiden nicht zwischen „Nur zu Besuch“ und „im Gefängnis“, und wir ignorieren die Regel, dass man einen Pasch braucht, um aus dem Gefängnis zu kommen, und nehmen stattdessen an, dass der Spieler bezahlt, um beim nächsten Zug herauszukommen.

Indem wir bei GO starten und die Felder von 00 bis 39 durchnummerieren, können wir diese zweistelligen Zahlen aneinanderhängen, um Strings zu erzeugen, die Sets von Feldern entsprechen.

Statistisch lässt sich zeigen, dass die drei beliebtesten Felder in der Reihenfolge JAIL (6,24 %) = Feld 10, E3 (3,18 %) = Feld 24 und GO (3,09 %) = Feld 00 sind. Diese drei beliebtesten Felder können mit dem sechsstelligen Modus-String `102400` dargestellt werden.

Wenn statt zwei sechsseitiger Würfel zwei `n`-seitige Würfel verwendet werden, finden Sie den sechsstelligen Modus-String.

```js
const GO = 0;
const JAIL = 10;
const GO_TO_JAIL = 30;

const C1 = 11;
const E3 = 24;
const H2 = 39;

const R1 = 5;
const R2 = 15;
const R3 = 25;

const U1 = 12;
const U2 = 28;

const SPECIAL_CARDS = 16;
const GAME_SQUARES = 40;

const CC1 = 2;
const CC2 = 17;
const CC3 = 33;
const CHESTS = [CC1, CC2, CC3];
const chestCardsMoves = [GO, JAIL];

const CH1 = 7;
const CH2 = 22;
const CH3 = 36;
const CHANCES = [CH1, CH2, CH3];
const chanceCardsMoves = [GO, JAIL, C1, E3, H2, R1];
const chanceToRailroad = { [CH1]: R2, [CH2]: R3, [CH3]: R1 };
const chanceToUtility = { [CH1]: U1, [CH2]: U2, [CH3]: U1 };

function multiplyMatrix(matrix1, matrix2) {
  const multiplied = [];

  for (let row = 0; row < matrix1.length; row++) {
    const newRow = [];
    for (let col = 0; col < matrix1[row].length; col++) {
      let newCell = 0;
      for (let i = 0; i < matrix1[row].length; i++) {
        const value1 = matrix1[row][i];
        const value2 = matrix2[i][col];
        newCell += value1 * value2;
      }
      newRow.push(newCell);
    }
    multiplied.push(newRow);
  }
  return multiplied;
}

function normalizeRow(row) {
  const sum = row.reduce((total, value) => total + value, 0);
  if (sum > 0) {
    for (let j = 0; j < row.length; j++) {
      const value = row[j];
      row[j] = value / sum;
    }
  }
}

function sortByProbability(board) {
  return board
    .map((probability, squareNo) => [squareNo, probability])
    .sort((a, b) => a[1] - b[1])
}

function getTopThree(board) {
  return sortByProbability(board)
    .slice(-3)
    .reverse()
    .map(([squareNo, _]) => squareNo.toString().padStart(2, '0')
    )
    .join('');
}

function didConverge(matrix1, matrix2, precision) {
  return matrix1.every((row, rowNo) => row.every((value1, colNo) => Math.abs(value1 - matrix2[rowNo][colNo]) <= precision))
}

function monopolyOdds(diceSides) {
  // Based on https://github.com/ByteThisCoding/project-euler/blob/master/problems/0084/0084.ts

  const timesRolled = new Array(diceSides * 2 + 1).fill(0);
  for (let dice1 = 1; dice1 <= diceSides; dice1++) {
    for (let dice2 = 1; dice2 <= diceSides; dice2++) {
      timesRolled[dice1 + dice2]++;
    }
  }

  // Transitions matrix contain probabilities of reaching each square (row values)
  // from each starting square (row no.).
  let transitions = [];
  for (let startSquare = 0; startSquare < GAME_SQUARES; startSquare++) {
    const row = new Array(GAME_SQUARES).fill(0);
    for (let rollResult = 2; rollResult <= diceSides * 2; rollResult++) {
      const rollChance = timesRolled[rollResult]
      const position = (startSquare + rollResult) % GAME_SQUARES;

      if (CHANCES.includes(position)) {
        // Chance cards ordering movement.
        for (let i = 0; i < chanceCardsMoves.length; i++) {
          const nextSquare = chanceCardsMoves[i];
          row[nextSquare] += rollChance / SPECIAL_CARDS;
        }
        row[chanceToRailroad[position]] += 2 * rollChance / SPECIAL_CARDS;
        row[chanceToUtility[position]] += rollChance / SPECIAL_CARDS;
        row[position - 3] += rollChance / SPECIAL_CARDS;

        // Rest non-moving Chance cards.
        row[position] += (SPECIAL_CARDS - chanceCardsMoves.length) * rollChance / SPECIAL_CARDS;
      } else if (CHESTS.includes(position)) {
        // Community Chest cards ordering movement.
        for (let i = 0; i < chestCardsMoves.length; i++) {
          const nextSquare = chestCardsMoves[i];
          row[nextSquare] += rollChance / SPECIAL_CARDS;
        }
        // Rest non-moving Community Chest cards.
        row[position] += (SPECIAL_CARDS - chestCardsMoves.length) * rollChance / SPECIAL_CARDS
      } else if (position === GO_TO_JAIL) {
        row[JAIL] += rollChance;
      } else {
        row[position] += rollChance;
      }
    }
    normalizeRow(row)
    transitions.push(row);
  }

  const precision = 0.000001;
  for (let i = 0; i < GAME_SQUARES; i++) {
    const next = multiplyMatrix(transitions, transitions);
    if (didConverge(transitions, next, precision)) {
      break;
    }
    transitions = next;
  }

  // All rows converge to the same values.
  return getTopThree(transitions[0]);
}
```
