# --description--

Nambari kamilifu ni nambari ambayo jumla ya vigawanyo vyake halisi ni sawa kabisa na nambari hiyo. Kwa mfano, jumla ya vigawanyo halisi vya 28 itakuwa 1 + 2 + 4 + 7 + 14 = 28, ambayo ina maana kwamba 28 ni nambari kamilifu.

Nambari `n` huitwa duni ikiwa jumla ya vigawanyo vyake halisi ni chini ya `n` na huitwa nyingi ikiwa jumla hii inazidi `n`.

Kwa kuwa 12 ni nambari nyingi ndogo zaidi, 1 + 2 + 3 + 4 + 6 = 16, nambari ndogo zaidi inayoweza kuandikwa kama jumla ya nambari mbili nyingi ni 24. Kwa uchambuzi wa kihesabu, inaonyeshwa kwamba nambari kamili zote kubwa kuliko 28123 zinaweza kuandikwa kama jumla ya nambari mbili nyingi. Hata hivyo, kikomo hiki cha juu hakiwezi kupunguzwa zaidi kwa uchambuzi ingawa inajulikana kwamba nambari kubwa zaidi ambayo haiwezi kuonyeshwa kama jumla ya nambari mbili nyingi ni chini ya kikomo hiki.

Tafuta jumla ya nambari kamili chanya zote &lt;= `n` ambazo haiwezi kuandikwa kama jumla ya nambari mbili nyingi.

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
