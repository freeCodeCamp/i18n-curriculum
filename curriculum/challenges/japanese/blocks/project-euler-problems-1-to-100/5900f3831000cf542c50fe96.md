# --description--

完全数とは、その真の約数の和がちょうどその数自身に等しい数のことです。例えば、28の真の約数の和は1 + 2 + 4 + 7 + 14 = 28となり、28は完全数であることがわかります。

数 `n` は、その真の約数の和が `n` より小さい場合は不足数と呼ばれ、この和が `n` を超える場合は過剰数と呼ばれます。

12は最小の過剰数であり、1 + 2 + 3 + 4 + 6 = 16となります。したがって、2つの過剰数の和として表せる最小の数は24です。数学的な解析により、28123より大きいすべての整数は2つの過剰数の和として表せることが示されています。しかし、この上限はこれ以上解析によって減らすことはできません。なお、2つの過剰数の和として表せない最大の数はこの上限より小さいことが知られています。

2つの過剰数の和として表せない、`n` 以下のすべての正の整数の和を求めてください。

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
