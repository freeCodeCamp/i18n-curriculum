# --description--

グーゴル（$10^{100}$）は非常に大きな数で、1の後に100個のゼロが続きます。$100^{100}$はほとんど想像できないほど大きく、1の後に200個のゼロが続きます。それらの大きさにもかかわらず、各数の桁の和はわずか1です。

`a`、`b` &lt; `n` の形の自然数 $a^b$ を考えたとき、最大の桁の和はいくつでしょうか？

```js
function powerfulDigitSum(n) {
  function sumDigitsOfPower(numA, numB) {
    let digitsSum = 0;
    let number = power(numA, numB);
    while (number > 0n) {
      const digit = number % 10n;
      digitsSum += parseInt(digit, 10);
      number = number / 10n;
    }
    return digitsSum;
  }

  function power(numA, numB) {
    let sum = 1n;
    for (let b = 0; b < numB; b++) {
      sum = sum * BigInt(numA);
    }
    return sum;
  }

  const limit = n - 1;
  let maxDigitsSum = 0;
  for (let a = limit; a > 0; a--) {
    for (let b = limit; b > 0; b--) {
      const curDigitSum = sumDigitsOfPower(a, b);
      if (curDigitSum > maxDigitsSum) {
        maxDigitsSum = curDigitSum;
      }
    }
  }
  return maxDigitsSum;
}
```
