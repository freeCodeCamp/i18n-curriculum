# --description--

반지름이 같은 세 개의 원이 더 큰 원 안에 배치되어 각 원 쌍이 서로 접하고 내부 원들이 겹치지 않도록 합니다. 네 개의 빈 "틈"이 있으며, 이 틈들은 반복적으로 더 많은 접하는 원들로 채워집니다.

<img alt="겹치지 않는 원들의 도표" src="https://cdn-media-1.freecodecamp.org/project-euler/199-circles-in-circles.gif" style="background-color: white; padding: 10px; display: block; margin-right: auto; margin-left: auto; margin-bottom: 1.2rem;">

각 반복마다, 각 틈에 최대 크기의 원이 배치되어 다음 반복을 위한 더 많은 틈이 만들어집니다. 3회 반복 후(그림 참조), 틈은 108개이고 원으로 덮이지 않은 영역의 분수는 소수점 여덟 자리까지 반올림하여 0.06790342입니다.

`n`회 반복 후 원으로 덮이지 않은 영역의 분수는 얼마입니까? 답은 x.xxxxxxxx 형식으로 소수점 여덟 자리까지 반올림하여 제출하세요.

function iterativeCirclePacking(n) {
  let k1 = 1;
  let k0 = k1 * (3 - 2 * Math.sqrt(3));
  let a0 = 1 / (k0 * k0);
  let a1 = 3 / (k1 * k1);
  a1 += 3 * getArea(k0, k1, k1, n);
  a1 += getArea(k1, k1, k1, n);
  let final = ((a0 - a1) / a0).toFixed(8);

  return parseFloat(final);
  function getArea(k1, k2, k3, depth) {
      if (depth == 0) return 0.0;
      let k4 = k1 + k2 + k3 + 2 * Math.sqrt(k1 * k2 + k2 * k3 + k3 * k1);
      let a = 1 / (k4 * k4);
      a += getArea(k1, k2, k4, depth - 1);
      a += getArea(k2, k3, k4, depth - 1);
      a += getArea(k3, k1, k4, depth - 1);
      return a;
  }
}
