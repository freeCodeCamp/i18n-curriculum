# --description--

同じ半径の3つの円が、より大きな円の内側に配置され、それぞれの円のペアが接し、内側の円同士は重ならないようになっています。4つの覆われていない「隙間」があり、そこにさらに接する円を反復的に詰めていきます。

<img alt="重ならない円の図" src="https://cdn-media-1.freecodecamp.org/project-euler/199-circles-in-circles.gif" style="background-color: white; padding: 10px; display: block; margin-right: auto; margin-left: auto; margin-bottom: 1.2rem;">

各反復で、隙間に最大サイズの円が配置され、次の反復のための隙間がさらに増えます。3回の反復後（図の通り）、隙間は108個になり、円で覆われていないエリアの分数は0.06790342（小数点以下8桁に丸めた値）です。

`n`回の反復後、円で覆われていないエリアの分数はいくらですか？答えは小数点以下8桁で x.xxxxxxxx の形式で答えてください。

function iterativeCirclePacking(n) {
  let k1 = 1;
  let k0 = k1 * (3 - 2 * Math.sqrt(3));
  let a0 = 1 / (k0 * k0);
  let a1 = 3 / (k1 * k1);
  a1 += 3 * getArea(k0, k1, k1, n);
  a1 += getArea(k1, k1, k1, n);
  let final = ((a0 - a1) / a0).toFixed(8);

  return parseFloat(final);
  function getArea(k1, k2, k3, depth) {
      if (depth == 0) return 0.0;
      let k4 = k1 + k2 + k3 + 2 * Math.sqrt(k1 * k2 + k2 * k3 + k3 * k1);
      let a = 1 / (k4 * k4);
      a += getArea(k1, k2, k4, depth - 1);
      a += getArea(k2, k3, k4, depth - 1);
      a += getArea(k3, k1, k4, depth - 1);
      return a;
  }
}
