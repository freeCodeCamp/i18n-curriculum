# --description--

تعرف المتتالية التكرارية التالية لمجموعة الأعداد الصحيحة الموجبة:

<div style='padding-left: 4em;'><var>n</var> → <var>n</var>/2 (<var>n</var> عدد زوجي)</div>

<div style='padding-left: 4em;'><var>n</var> → 3<var>n</var> + 1 (<var>n</var> عدد فردي)</div>

باستخدام القاعدة أعلاه والبدء بالعدد 13، نولد المتتالية التالية:

<div style='text-align: center;'>13 → 40 → 20 → 10 → 5 → 16 → 8 → 4 → 2 → 1</div>

يمكن ملاحظة أن هذه المتتالية (التي تبدأ عند 13 وتنتهي عند 1) تحتوي على 10 حدود. رغم أنه لم يُثبت بعد (مشكلة كولاتز)، يُعتقد أن جميع الأعداد الابتدائية تنتهي عند 1.

أي عدد ابتدائي، تحت `limit` المعطى، ينتج أطول سلسلة؟

**ملاحظة:** بمجرد بدء السلسلة، يُسمح للحدود أن تتجاوز `limit`.

function longestCollatzSequence(limit) {
  let longest = 1;
  let maxLength = 1;
  for (let i = Math.floor(limit / 2); i < limit; i++) {
    let len = colLen(i);
    if (len > maxLength) {
      longest = i;
      maxLength = len;
    }
  }
  return longest;
}

const knownSequence = { '1': 1 };

function colLen(n) {
  if (knownSequence[n]) {
    return knownSequence[n];
  } else {
    const len = n % 2 === 0 ? colLen(n / 2) + 1 : colLen((3 * n + 1) / 2) + 2;
    knownSequence[n] = len;
    return len;
  }
}
