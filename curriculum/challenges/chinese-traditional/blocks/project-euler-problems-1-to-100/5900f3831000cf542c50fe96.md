# --description--

完美數是指其真因數的總和恰好與該數字相等的數字。 例如，28 的真因數總和為 1 + 2 + 4 + 7 + 14 = 28，這表示 28 是一個完美數。

當一個數字 `n` 的真因數和小於 `n` 時，稱該數字為虧數；當該和大於 `n` 時，稱為盈數。

由於 12 是最小的豐裕數，1 + 2 + 3 + 4 + 6 = 16，能寫成兩個豐裕數之和的最小數字是 24。透過數學分析，可以證明所有大於 28123 的整數都可以寫成兩個豐裕數之和。然而，儘管已知不能表示為兩個豐裕數之和的最大數字小於此上限，但此上限無法再透過分析進一步降低。

找出所有小於等於 `n` 且無法寫成兩個豐富數之和的正整數的總和。

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
