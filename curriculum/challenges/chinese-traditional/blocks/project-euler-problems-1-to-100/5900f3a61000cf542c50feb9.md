# --description--

從 1 開始，並以逆時針方向螺旋排列，形成邊長為 7 的方形螺旋。

有趣的是，奇數平方數位於右下對角線上，但更有趣的是，沿著兩條對角線的 13 個數字中有 8 個是質數；也就是說，比例為 8/13 ≈ 62%。

如果在上方的螺旋周圍包覆一整層新的層，將形成邊長為 9 的方形螺旋。若繼續此處理程序，當兩條對角線上的質數百分比首次低於 `percent` 時，該方形螺旋的邊長為多少？

function spiralPrimes(percent) {
  function isPrime(n) {
    if (n <= 3) {
      return n > 1;
    } else if (n % 2 === 0 || n % 3 === 0) {
      return false;
    }

    for (let i = 5; i * i <= n; i += 6) {
      if (n % i === 0 || n % (i + 2) === 0) {
        return false;
      }
    }
    return true;
  }

  let totalCount = 1;
  let primesCount = 0;
  let curNumber = 1;
  let curSideLength = 1;
  let ratio = 1;
  const wantedRatio = percent / 100;

  while (ratio >= wantedRatio) {
    curSideLength += 2;
    for (let i = 0; i < 4; i++) {
      curNumber += curSideLength - 1;
      totalCount++;
      if (i !== 3 && isPrime(curNumber)) {
        primesCount++;
      }
    }
    ratio = primesCount / totalCount;
  }
  return curSideLength;
}
