# --description--

Tres círculos de radio igual se colocan dentro de un círculo más grande de manera que cada par de círculos sea tangente entre sí y los círculos internos no se superpongan. Hay cuatro "espacios" descubiertos que se deben llenar de forma iterativa con más círculos tangentes.

<img alt="a diagram of non-overlapping circles" src="https://cdn-media-1.freecodecamp.org/project-euler/199-circles-in-circles.gif" style="background-color: white; padding: 10px; display: block; margin-right: auto; margin-left: auto; margin-bottom: 1.2rem;">

En cada iteración, se coloca un círculo de tamaño máximo en cada espacio, lo que crea más espacios para la siguiente iteración. Después de 3 iteraciones (en la imagen), hay 108 espacios y la fracción del área que no está cubierta por círculos es 0.06790342, redondeada a ocho decimales.

¿Qué fracción del área no está cubierta por círculos después de `n` iteraciones? Da tu respuesta redondeada a ocho decimales usando el formato x.xxxxxxxx .

function iterativeCirclePacking(n) {
  let k1 = 1;
  let k0 = k1 * (3 - 2 * Math.sqrt(3));
  let a0 = 1 / (k0 * k0);
  let a1 = 3 / (k1 * k1);
  a1 += 3 * getArea(k0, k1, k1, n);
  a1 += getArea(k1, k1, k1, n);
  let final = ((a0 - a1) / a0).toFixed(8);

  return parseFloat(final);
  function getArea(k1, k2, k3, depth) {
      if (depth == 0) return 0.0;
      let k4 = k1 + k2 + k3 + 2 * Math.sqrt(k1 * k2 + k2 * k3 + k3 * k1);
      let a = 1 / (k4 * k4);
      a += getArea(k1, k2, k4, depth - 1);
      a += getArea(k2, k3, k4, depth - 1);
      a += getArea(k3, k1, k4, depth - 1);
      return a;
  }
}
