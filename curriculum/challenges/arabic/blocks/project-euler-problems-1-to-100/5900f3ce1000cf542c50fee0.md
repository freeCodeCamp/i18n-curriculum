# --description--

تم اكتشاف أول عدد أولي معروف يتجاوز مليون رقم في عام 1999، وهو عدد أولي ميرسين من الشكل $2^{6972593} − 1$؛ يحتوي بالضبط على 2,098,960 رقمًا. بعد ذلك، تم العثور على أعداد أولية ميرسين أخرى من الشكل $2^p − 1$ تحتوي على أرقام أكثر.

ومع ذلك، في عام 2004 تم العثور على عدد أولي ضخم غير ميرسين يحتوي على 2,357,207 رقمًا: $28433 × 2^{7830457} + 1$.

اعثر على آخر عشرة أرقام لذلك العدد الأولي غير ميرسين بالشكل $multiplier × 2^{power} + 1$.

function largeNonMersennePrime(multiplier, power) {
  function modStepsResults(number, other, mod, startValue, step) {
    let result = startValue;
    for (let i = 0; i < other; i++) {
      result = step(number, result) % mod;
    }
    return result;
  }

  const numOfDigits = 10;
  const mod = 10 ** numOfDigits;
  const digitsAfterPower = modStepsResults(2, power, mod, 1, (a, b) => a * b);
  const digitsAfterMultiply = modStepsResults(
    digitsAfterPower,
    multiplier,
    mod,
    0,
    (a, b) => a + b
  );
  const lastDigits = (digitsAfterMultiply + 1) % mod;

  return lastDigits.toString().padStart(10, '0');
}
