# --description--

숫자 1406357289는 0부터 9까지의 범숫자의로, 0부터 9까지의 각 숫자가 어떤 순서로든 모두 포함되어 있지만, 꽤 흥미로운 부분 문자열 나눗셈 속성도 가지고 있습니다.

$d_1$을 첫 번째 숫자, $d_2$를 두 번째 숫자, 이런 식으로 정의하면 다음과 같습니다:

- ${d_2}{d_3}{d_4} = 406$는 2로 나누어떨어집니다
- ${d_3}{d_4}{d_5} = 063$는 3으로 나누어떨어집니다
- ${d_4}{d_5}{d_6} = 635$는 5로 나누어떨어집니다
- ${d_5}{d_6}{d_7} = 357$는 7로 나누어떨어집니다
- ${d_6}{d_7}{d_8} = 572$는 11로 나누어떨어집니다
- ${d_7}{d_8}{d_9} = 728$는 13으로 나누어떨어집니다
- ${d_8}{d_9}{d_{10}} = 289$는 17로 나누어떨어집니다

이러한 나눗셈 속성 중 `n`개를 만족하는 부분 문자열을 가진 0부터 `n - 2`까지의 모든 범숫자의의 합을 구하세요.

**참고:** `0`로 시작하는 범숫자의도 결과에 포함하세요.

function substringDivisibility(n) {
  function isSubDivisible(digits) {
    const factors = [2, 3, 5, 7, 11, 13, 17];

    for (let i = 1; i < digits.length - 2; i++) {
      const subNumber = digits[i] * 100 + digits[i + 1] * 10 + digits[i + 2];
      if (subNumber % factors[i - 1] !== 0) {
        return false;
      }
    }
    return true;
  }

  function heapsPermutations(k, digits, conditionCheck, results) {
    if (k === 1) {
      if (conditionCheck(digits)) {
        const number = parseInt(digits.join(''), 10);
        results.push(number);
      }
      return;
    }

    heapsPermutations(k - 1, digits, conditionCheck, results);

    for (let i = 0; i < k - 1; i++) {
      if (k % 2 === 0) {
        [digits[i], digits[k - 1]] = [digits[k - 1], digits[i]];
      } else {
        [digits[0], digits[k - 1]] = [digits[k - 1], digits[0]];
      }
      heapsPermutations(k - 1, digits, conditionCheck, results);
    }
    return;
  }

  const allowedDigits = [...new Array(n + 1).keys()];
  const divisiblePandigitals = [];
  heapsPermutations(
    allowedDigits.length,
    allowedDigits,
    isSubDivisible,
    divisiblePandigitals
  );

  let sum = 0;
  for (let i = 0; i < divisiblePandigitals.length; i++) {
    sum += divisiblePandigitals[i];
  }

  return sum;
}
