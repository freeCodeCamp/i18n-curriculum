# --description--

1から始めて反時計回りに螺旋を描くと、辺の長さが7の正方形の螺旋が形成されます。

この螺旋の周りに新しい層を1つ追加すると、辺の長さが9の正方形の螺旋が形成されます。このプロセスを続けた場合、両対角線上の素数の割合が初めて`percent`を下回る正方形の螺旋の辺の長さはいくつでしょうか？

function spiralPrimes(percent) {
  function isPrime(n) {
    if (n <= 3) {
      return n > 1;
    } else if (n % 2 === 0 || n % 3 === 0) {
      return false;
    }

    for (let i = 5; i * i <= n; i += 6) {
      if (n % i === 0 || n % (i + 2) === 0) {
        return false;
      }
    }
    return true;
  }

  let totalCount = 1;
  let primesCount = 0;
  let curNumber = 1;
  let curSideLength = 1;
  let ratio = 1;
  const wantedRatio = percent / 100;

  while (ratio >= wantedRatio) {
    curSideLength += 2;
    for (let i = 0; i < 4; i++) {
      curNumber += curSideLength - 1;
      totalCount++;
      if (i !== 3 && isPrime(curNumber)) {
        primesCount++;
      }
    }
    ratio = primesCount / totalCount;
  }
  return curSideLength;
}
