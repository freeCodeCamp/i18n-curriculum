    n = len(matrix)
    distances = [INF] * n

    visited = [False] * n

        min_distance = INF
        current = -1
        for node_no in range(n):
            if not visited[node_no] and distances[node_no] < min_distance:
                min_distance = distances[node_no]
                current = node_no

                    paths[node_no] = paths[current] + [node_no]
