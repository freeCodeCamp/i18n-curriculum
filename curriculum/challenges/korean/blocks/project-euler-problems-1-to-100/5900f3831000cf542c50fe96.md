# --description--

완전수는 그 진약수의 합이 정확히 자기 자신과 같은 수입니다. 예를 들어, 28의 진약수 합은 1 + 2 + 4 + 7 + 14 = 28이므로 28은 완전수입니다.

수 `n`는 진약수의 합이 `n`보다 작으면 부족수라고 하고, 이 합이 `n`를 초과하면 과잉수라고 합니다.

12는 가장 작은 과잉수로, 1 + 2 + 3 + 4 + 6 = 16입니다. 두 과잉수의 합으로 쓸 수 있는 가장 작은 수는 24입니다. 수학적 분석에 따르면 28123보다 큰 모든 정수는 두 과잉수의 합으로 나타낼 수 있습니다. 하지만 두 과잉수의 합으로 나타낼 수 없는 가장 큰 수가 이 한계보다 작다는 사실에도 불구하고, 이 상한선을 더 낮출 수는 없습니다.

두 과잉수의 합으로 나타낼 수 없는 모든 양의 정수 중 `n` 이하인 수의 합을 구하세요.

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
