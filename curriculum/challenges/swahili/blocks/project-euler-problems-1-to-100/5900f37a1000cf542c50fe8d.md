<div style='padding-left: 4em;'><var>n</var> → <var>n</var>/2 (<var>n</var> ni nambari kamili chanya)</div>

<div style='text-align: center;'>13 → 40 → 20 → 10 → 5 → 16 → 8 → 4 → 2 → 1</div>

Ni nambari gani ya kuanzia, chini ya `limit`, huunda mnyororo mrefu zaidi?

**Kumbuka:** Mara mnyororo unapoanza, maneno yanaruhusiwa kwenda juu ya `limit`.

function longestCollatzSequence(limit) {
  let longest = 1;
  let maxLength = 1;
  for (let i = Math.floor(limit / 2); i < limit; i++) {
    let len = colLen(i);
    if (len > maxLength) {
      longest = i;
      maxLength = len;
    }
  }
  return longest;
}

const knownSequence = { '1': 1 };

function colLen(n) {
  if (knownSequence[n]) {
    return knownSequence[n];
  } else {
    const len = n % 2 === 0 ? colLen(n / 2) + 1 : colLen((3 * n + 1) / 2) + 2;
    knownSequence[n] = len;
    return len;
  }
}
