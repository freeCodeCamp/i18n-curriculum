# --description--

Напишіть генератор простих чисел у порядку зростання, який автоматично підлаштовується для генерації будь-якого розумно великого простого числа.

Генератор має вміти:

<ul>
  <li>Показувати перші <code>n</code> простих чисел</li>
  <li>Показувати прості числа в діапазоні</li>
  <li>Показувати кількість простих чисел у діапазоні</li>
  <li>Показувати <code>n<sup>те</sup></code> просте число</li>
</ul>

Функція має мати два параметри. Перший прийматиме `n` або діапазон у вигляді масиву. Другий прийматиме булевий тип, який вказує, чи функція повертає прості числа у вигляді масиву, чи одне число (кількість простих чисел у діапазоні або <code>n<sup>те</sup></code> просте число). Залежно від параметрів функція має повертати масив.

```js
function primeGenerator(num, showPrimes) {
  let i,
    arr = [];

  function isPrime(num) {
    // try primes <= 16
    if (num <= 16) { return (
      num == 2 || num == 3 || num == 5 || num == 7 || num == 11 || num == 13
    ); }
    // cull multiples of 2, 3, 5 or 7
    if (num % 2 == 0 || num % 3 == 0 || num % 5 == 0 || num % 7 == 0)
      { return false; }
    // cull square numbers ending in 1, 3, 7 or 9
    for (let i = 10; i * i <= num; i += 10) {
      if (num % (i + 1) == 0) return false;
      if (num % (i + 3) == 0) return false;
      if (num % (i + 7) == 0) return false;
      if (num % (i + 9) == 0) return false;
    }
    return true;
  }

  if (typeof num === 'number') {
    for (i = 0; arr.length < num; i++) if (isPrime(i)) arr.push(i);
    // first x primes
    if (showPrimes) return arr;
    // xth prime
    return arr.pop();
  }

  if (Array.isArray(num)) {
    for (i = num[0]; i <= num[1]; i++) if (isPrime(i)) arr.push(i);
    // primes between x .. y
    if (showPrimes) return arr;
    // number of primes between x .. y
    return arr.length;
  }
}
```
