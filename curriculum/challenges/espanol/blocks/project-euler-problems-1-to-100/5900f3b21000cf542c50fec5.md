# --description--

La función totiente de Euler, ${\phi}(n)$ (a veces llamada función phi), se usa para determinar la cantidad de números positivos menores o iguales a `n` que son primos relativos con `n`. Por ejemplo, como 1, 2, 4, 5, 7 y 8 son todos menores que nueve y primos relativos con nueve, ${\phi}(9) = 6$. El número 1 se considera primo relativo con todos los números positivos, por lo que ${\phi}(1) = 1$.

Curiosamente, ${\phi}(87109) = 79180$, y se puede ver que 87109 es una permutación de 79180.

Encuentra el valor de `n`, 1 &lt; `n` &lt; `limit`, para el cual ${\phi}(n)$ es una permutación de `n` y la razón $\displaystyle\frac{n}{{\phi}(n)}$ produce un mínimo.

function totientPermutation(limit) {
  function getSievePrimes(max) {
    const primes = [];
    const primesMap = new Array(max).fill(true);
    primesMap[0] = false;
    primesMap[1] = false;

    for (let i = 2; i < max; i += 2) {
      if (primesMap[i]) {
        primes.push(i);
        for (let j = i * i; j < max; j += i) {
          primesMap[j] = false;
        }
      }
      if (i === 2) {
        i = 1;
      }
    }
    return primes;
  }

  function sortDigits(number) {
    return number.toString().split('').sort().join('');
  }

  function isPermutation(numberA, numberB) {
    return sortDigits(numberA) === sortDigits(numberB);
  }

  const MAX_PRIME = 4000;
  const primes = getSievePrimes(MAX_PRIME);

  let nValue = 1;
  let minRatio = Infinity;

  for (let i = 1; i < primes.length; i++) {
    for (let j = i + 1; j < primes.length; j++) {
      const num = primes[i] * primes[j];
      if (num > limit) {
        break;
      }

      const phi = (primes[i] - 1) * (primes[j] - 1);
      const ratio = num / phi;

      if (minRatio > ratio && isPermutation(num, phi)) {
        nValue = num;
        minRatio = ratio;
      }
    }
  }
  return nValue;
}
