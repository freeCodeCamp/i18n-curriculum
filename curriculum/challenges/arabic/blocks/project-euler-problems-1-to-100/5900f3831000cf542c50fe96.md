# --description--

العدد الكامل هو عدد يكون مجموع قواسمه الصحيحة بالضبط مساويًا له. على سبيل المثال، مجموع القواسم الصحيحة للعدد 28 هو 1 + 2 + 4 + 7 + 14 = 28، مما يعني أن 28 هو عدد كامل.

يُسمى العدد `n` ناقصًا إذا كان مجموع قواسمه الصحيحة أقل من `n`، ويُسمى وفيرًا إذا تجاوز هذا المجموع `n`.

بما أن 12 هو أصغر عدد وفير، حيث 1 + 2 + 3 + 4 + 6 = 16، فإن أصغر عدد يمكن كتابته كمجموع عددين وفيرين هو 24. يمكن إثبات بالتحليل الرياضي أن جميع الأعداد الصحيحة الأكبر من 28123 يمكن كتابتها كمجموع عددين وفيرين. مع ذلك، لا يمكن تقليل هذا الحد الأعلى أكثر بالتحليل، رغم معرفة أن أكبر عدد لا يمكن التعبير عنه كمجموع عددين وفيرين أقل من هذا الحد.

احسب مجموع جميع الأعداد الصحيحة الموجبة &lt;= `n` التي لا يمكن كتابتها كمجموع عددين وفيرين.

function abundantCheck(number) {
  let sum = 1;

  for (let i = 2; i <= Math.sqrt(number); i += 1) {
    if(number % i === 0) {
      sum += i + +(i !== Math.sqrt(number) && number / i);
    }
  }
  return sum > number;
}

function sumOfNonAbundantNumbers(n) {
  let sum = 0;
  const memo = {};
  let abundantList = [];

  // Function checkSum checks if num can be represented as a sum of numbers in the stack (array)
  const checkSum = (num, stack, memo) => {
    for (let i = 0; i < stack.length; i += 1) {
      if ((num - stack[i]) in memo) return true;
    }
    return false;
  };

  for (let i = 1; i <= n; i += 1) {
    if (abundantCheck(i)) {
      abundantList.push(i);
      memo[i] = 1;
    }
    if (checkSum(i, abundantList, memo)) continue;
    sum += i;
  }
  return sum;
}
