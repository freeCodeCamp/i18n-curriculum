# --description--

لتكن دالة d(`n`) معرفة كمجموع القواسم الصحيحة لـ `n` (الأعداد الأقل من `n` التي تقسم `n` بدون باقي).

إذا كان d(`a`) = `b` و d(`b`) = `a`، حيث `a` ≠ `b`، فإن `a` و `b` يشكلان زوجًا وديًا، وكل من `a` و `b` يسمى عددًا وديًا.

على سبيل المثال، القواسم الصحيحة للعدد 220 هي 1، 2، 4، 5، 10، 11، 20، 22، 44، 55 و110؛ لذلك d(220) = 284. القواسم الصحيحة للعدد 284 هي 1، 2، 4، 71 و142؛ إذن d(284) = 220.

احسب مجموع كل الأعداد الودية الأقل من `n`.

const sumAmicableNum = (n) => {
  const fsum = (n) => {
    let sum = 1;
    for (let i = 2; i <= Math.floor(Math.sqrt(n)); i++)
      if (Math.floor(n % i) === 0)
        sum += i + Math.floor(n / i);
    return sum;
  };
  let d = [];
  let amicableSum = 0;
  for (let i=2; i<n; i++) d[i] = fsum(i);
  for (let i=2; i<n; i++) {
    let dsum = d[i];
    if (d[dsum]===i && i!==dsum) amicableSum += i+dsum;
  }
  return amicableSum/2;
};
