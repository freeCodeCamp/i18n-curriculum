#### --distractors--

```jsx
function handleSubmit(event) {
  event.defaultPrevent();
  console.log("Form submitted!");
}

  <input type="text" />
  <button>Submit</button>
</form>;
```

---

```jsx
function handleSubmit(event) {
  event.Default();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```

---

```jsx
function handleSubmit(event) {
  event.prevent();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```

#### --answer--


```jsx

  event.preventDefault();
  console.log("Form submitted!");
}

<form onSubmit={handleSubmit}>
  <input type="text" />
  <button>Submit</button>
</form>;
```
