# --description--

Nambari 145 inajulikana kwa sifa kwamba jumla ya faktoria za tarakimu zake ni sawa na 145:

$$1! + 4! + 5! = 1 + 24 + 120 = 145$$

Labda haijulikani sana ni 169, kwa sababu huunda mnyororo mrefu zaidi wa nambari unaorejea nyuma kwa 169; inajitokeza kuwa kuna mizunguko mitatu tu kama hiyo inayopatikana:

$$\begin{align}
&169 → 363601 → 1454 → 169\\\\
&871 → 45361 → 871\\\\
&872 → 45362 → 872\\\\
\end{align}$$

Si vigumu kuthibitisha kuwa KILA nambari ya kuanzia hatimaye itakwama katika mzunguko. Kwa mfano,

$$\begin{align}
&69 → 363600 → 1454 → 169 → 363601\\ (→ 1454)\\\\
&78 → 45360 → 871 → 45361\\ (→ 871)\\\\
&540 → 145\\ (→ 145)\\\\
\end{align}$$

Kuanzia na 69 huunda mnyororo wa maneno matano yasiyorudiwa, lakini mnyororo mrefu zaidi usiorudiwa wenye nambari ya kuanzia chini ya milioni moja ni maneno sitini.

Ni mnyororo mingapi, yenye nambari ya kuanzia chini ya `n`, ina maneno sitini yasiyorudiwa kabisa?

function digitFactorialChains(n) {
  function sumDigitsFactorials(number) {
    let sum = 0;
    while (number > 0) {
      sum += factorials[number % 10];
      number = Math.floor(number / 10);
    }
    return sum;
  }

  const factorials = [1];
  for (let i = 1; i < 10; i++) {
    factorials.push(factorials[factorials.length - 1] * i);
  }

  const sequences = {
    169: 3,
    871: 2,
    872: 2,
    1454: 3,
    45362: 2,
    45461: 2,
    3693601: 3
  };
  let result = 0;

  for (let i = 2; i < n; i++) {
    let curNum = i;
    let chainLength = 0;
    const curSequence = [];
    while (curSequence.indexOf(curNum) === -1) {
      curSequence.push(curNum);
      curNum = sumDigitsFactorials(curNum);
      chainLength++;
      if (sequences.hasOwnProperty(curNum) > 0) {
        chainLength += sequences[curNum];
        break;
      }
    }
    if (chainLength === 60) {
      result++;
    }
    for (let j = 1; j < curSequence.length; j++) {
      sequences[curSequence[j]] = chainLength - j;
    }
  }
  return result;
}
