# --description--

Wavuvi watano wameanguka baharini na kukutana kisiwa na kukusanya mzigo mkubwa wa nazi wakati wa mchana.

Usiku huo mvuvi wa kwanza anaamka na kuamua kuchukua sehemu yake ya kwanza mapema, hivyo anajaribu kugawanya mzigo wa nazi kwa usawa katika makundi matano lakini anagundua kuna nazi moja iliyobaki, hivyo anaitupa kwa tumbili kisha anaficha moja ya makundi matano ya nazi yenye ukubwa sawa na kusukuma makundi mengine manne pamoja tena kuunda mzigo mmoja unaoonekana wa nazi na anarudi kulala.

Ili kuufupisha hadithi, kila mvuvi kwa mfuatano anaamka mara moja usiku na kufanya vitendo sawa vya kugawanya mzigo wa nazi kwa makundi matano, kugundua kuna nazi moja iliyobaki na kumpa tumbili nazi hiyo moja iliyobaki, kisha kusukuma makundi manne pamoja kuunda mzigo mmoja.

Asubuhi (baada ya vitendo vya siri na vya pekee vya kila mvuvi wa tano usiku), nazi zilizobaki hugawanywa kwa makundi matano sawa kwa kila mvuvi, ndipo kugundulika mzigo wa nazi hugawanyika kwa usawa kati ya wavuvi bila mabaki. (Hakuna nazi kwa tumbili asubuhi.)

function splitCoconuts(intSailors) {
  let intNuts = intSailors;
  let result = splitCoconutsHelper(intNuts, intSailors);
  while (!result) {
    intNuts += 1;
    result = splitCoconutsHelper(intNuts, intSailors);
  }

  return intNuts;
}

function splitCoconutsHelper(intNuts, intSailors, intDepth) {
  const nDepth = intDepth !== undefined ? intDepth : intSailors;
  const portion = Math.floor(intNuts / intSailors);
  const remain = intNuts % intSailors;

  if (portion <= 0 || remain !== (nDepth ? 1 : 0)) {
    return null;
  }

  if (nDepth) {
    return splitCoconutsHelper(
      intNuts - portion - remain, intSailors, nDepth - 1
    );
  }

  return intNuts;
}
