# --description--

العدد المتناظر 595 مثير للاهتمام لأنه يمكن كتابته كمجموع لمربعات متتالية: $6^2 + 7^2 + 8^2 + 9^2 + 10^2 + 11^2 + 12^2$.

هناك بالضبط أحد عشر عددًا متناظرًا أقل من الألف يمكن كتابتها كمجموع مربعات متتالية، ومجموع هذه الأعداد المتناظرة هو 4164. لاحظ أن $1 = 0^2 + 1^2$ لم يُدرج لأن هذه المشكلة تهتم بمربعات الأعداد الصحيحة الموجبة.

احسب مجموع كل الأعداد الأقل من `limit` التي هي متناظرة ويمكن كتابتها كمجموع مربعات متتالية.

```js
function isPalindrome(num) {
  return num
    .toString()
    .split('')
    .every((digit, i, arr) => digit === arr[arr.length - 1 - i]);
}

function palindromicSums(limit) {
  let sumOfPalindromes = 0;
  const sqrtLimit = Math.sqrt(limit);
  const list = new Set();

  for (let i = 1; i <= sqrtLimit; i++) {
    let sumOfSquares = i * i;
    for (let j = i + 1; j <= sqrtLimit; j++) {
      sumOfSquares += j * j;
      if (sumOfSquares > limit) break;
      if (isPalindrome(sumOfSquares) && !list.has(sumOfSquares)) {
        sumOfPalindromes += sumOfSquares;
        list.add(sumOfSquares);
      }
    }
  }
  return sumOfPalindromes;
}
```
