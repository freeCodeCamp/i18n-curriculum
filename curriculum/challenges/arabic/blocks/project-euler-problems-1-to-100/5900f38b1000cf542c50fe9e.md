# --description--

في إنجلترا، تتكون العملة من الجنيه الإسترليني (£) والبنس (p)، وهناك ثمانية عملات متداولة بشكل عام:

<div style='margin-left: 4em;'>1p، 2p، 5p، 10p، 20p، 50p، £1 (100p) و£2 (200p).</div>

يمكن تكوين £2 بالطريقة التالية:

<div style='margin-left: 4em;'>1×£1 + 1×50p + 2×20p + 1×5p + 1×2p + 3×1p</div>

كم عدد الطرق المختلفة التي يمكن بها تكوين `n` بنس باستخدام أي عدد من العملات؟

```js
const coinSums = (n) => {
  const getWays = (n, m=8, c=[1, 2, 5, 10, 20, 50, 100, 200]) => {
    if (n === 0) return 1;
    if (m === 0 || n < 0) return 0;
    return getWays(n - c[m - 1], m, c) + getWays(n, m - 1, c);
  };
  return getWays(n);
};
```
